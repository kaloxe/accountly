SELECT account.name_account, name_badge, ((SELECT SUM(transaction.amount) FROM transaction WHERE transaction.id_account=5 and badge.id_badge=transaction.id_badge and transaction.type=1)-(SELECT SUM(transaction.amount) FROM transaction WHERE transaction.id_account=5 and badge.id_badge=transaction.id_badge and transaction.type=0)) FROM account INNER JOIN transaction on account.id_account=transaction.id_account INNER JOIN badge on transaction.id_badge=badge.id_badge WHERE account.id_account=5 GROUP BY badge.id_badge;
